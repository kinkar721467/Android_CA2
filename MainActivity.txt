@Composable
fun TicketBooking() {

    var noseats by remember { mutableStateOf("") }
    var paidAmount by remember { mutableStateOf("") }
    var name by remember { mutableStateOf("") }

    val context = LocalContext.current
    val ticketPrice = 500

    Column(
        modifier = Modifier.fillMaxSize()
            .background(Color.Cyan),
        verticalArrangement = Arrangement.Center,
        horizontalAlignment = Alignment.CenterHorizontally
    ) {
        Image(
            painter = painterResource(R.drawable.movie),
            contentDescription = "movie"
        )

        Text(
            text = "3 Idiots",
            fontSize = 22.sp
        )

        Text(
            text = "Ticket price: â‚¹500 per person",
            fontSize = 20.sp,
        )

        OutlinedTextField(
            value = name,
            onValueChange = { name = it },
            label = { Text("Name") }
        )

        OutlinedTextField(
            value = noseats,
            onValueChange = { noseats = it },
            label = { Text("No of seats") }
        )

        OutlinedTextField(
            value = paidAmount,
            onValueChange = { paidAmount = it },
            label = { Text("Paid amount") }
        )

        Button(
            onClick = {
                val seats = noseats.toIntOrNull() ?: 0
                val amount = paidAmount.toIntOrNull() ?: 0

                when {
                    name.isEmpty() -> {
                        Toast.makeText(context, "Name is empty", Toast.LENGTH_LONG).show()
                    }

                    seats <= 0 -> {
                        Toast.makeText(context, "Enter valid number of seats", Toast.LENGTH_LONG).show()
                    }

                    amount != seats * ticketPrice -> {
                        Toast.makeText(context, "Incorrect amount", Toast.LENGTH_LONG).show()
                    }

                    else -> {
                        Toast.makeText(context, "Movie ticket booked successfully", Toast.LENGTH_LONG).show()

                        val intent = Intent(Intent.ACTION_SEND).apply {
                            type = "text/plain"
                            putExtra(Intent.EXTRA_TEXT, "Movie ticket booked successfully")
                        }
                        context.startActivity(intent)
                    }
                }
            }
        ) {
            Text("Book Ticket")
        }
    }
}
